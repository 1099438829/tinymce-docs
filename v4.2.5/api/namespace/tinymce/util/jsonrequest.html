--- layout: default title: tinymce.util.JSONRequest ---
This class enables you to use JSON-RPC to call backend methods.

Examples

var json = new tinymce.util.JSONRequest({
    url: 'somebackend.php'
});

// Send RPC call 1
json.send({
    method: 'someMethod1',
    params: ['a', 'b'],
    success: function(result) {
        console.dir(result);
    }
});

// Send RPC call 2
json.send({
    method: 'someMethod2',
    params: ['a', 'b'],
    success: function(result) {
        console.dir(result);
    }
});
